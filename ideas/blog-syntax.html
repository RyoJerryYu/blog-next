class HelloPrinter:
    printer: Callable[[str]]
 
    def __init__(self, printer: Callable[[str]]):
        self.printer = printer
    
    def call(self, s: str):
        self.printer(s)
 
 
p = HelloPrinter(func_echo)
p.call("hello world!")
